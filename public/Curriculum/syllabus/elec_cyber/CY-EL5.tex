\section*{Mobile \& Wireless Security (CY-EL5)}
\label{elec:CY_5}

\noindent \small{\hyperref[main:scheme]{$\leftarrow$ Back to Scheme}} \vspace{0.5cm}

\noindent \textbf{Credits:} 4 (3-0-2)

\subsection*{Theory}
\noindent\textbf{Unit 1} \\
\textit{Mobile Platform Security Architecture:} Android security model (sandboxing, app permissions, SELinux), iOS security (code signing, sandbox, XNU kernel), App lifecycle and inter-process communication (Intents, AIDL, XPC), Secure boot chain and verified boot, Mobile device management (MDM) frameworks, Root/jailbreak detection bypasses, Hardware security (TEE, ARM TrustZone).

\vspace{0.3cm}

\noindent\textbf{Unit 2} \\
\textit{Mobile Application Security:} Static analysis (MobSF, QARK), Dynamic analysis (Frida, Objection), Reverse engineering (APKTool, Hopper, IDA), Common vulnerabilities (insecure storage, hard-coded secrets, improper IPC), OWASP Mobile Top 10, Certificate pinning bypass, Runtime hooking and SSL pinning evasion, Binary protection (obfuscation, RASP).

\vspace{0.3cm}

\noindent\textbf{Unit 3} \\
\textit{Wireless Network Security Protocols:} 802.11 wireless fundamentals (WEP, WPA, WPA2, WPA3), Wi-Fi attack vectors (evil twin, KRACK, PMKID), Enterprise security (802.1X/EAP methods: PEAP, EAP-TLS), Bluetooth Low Energy (BLE) security (pairing modes, Just Works vulnerability), BLE attacks (knob turning, spoofing), Zigbee/IoT protocol security, Wireless IDS/IPS.

\vspace{0.3cm}

\noindent\textbf{Unit 4} \\
\textit{Cellular Network Security:} GSM/CDMA vulnerabilities (A5/1 cracking, IMSI catchers), LTE/4G security (EPS-AKA, SUCI encryption), 5G security architecture (SUPI concealment, SEAF authentication), Mobile network attacks (SS7 exploitation, Diameter signaling), IMS/SIP security, Roaming security risks, SIM toolkit and OTA attacks, eSIM security.

\vspace{0.3cm}

\noindent\textbf{Unit 5} \\
\textit{Mobile Threat Landscape and Forensics:} Mobile malware evolution (banking trojans, spyware), Phishing and social engineering vectors, MDM bypass and enterprise compromise, Mobile forensics (logical, file system, physical extraction), JTAG/ISP chip-off methods, Anti-forensics (app shredders, secure erase), Mobile security frameworks (MASA, GSMA NESAS).

\subsection*{Practical}
\noindent\textbf{Lab 1: Android App Reverse Engineering} \\
Focuses on APK decompilation, manifest analysis, permission abuse.

\vspace{0.2cm}

\noindent\textbf{Lab 2: iOS App Analysis} \\
Focuses on IPA cracking, runtime hooking with Frida/Cycript.

\vspace{0.2cm}

\noindent\textbf{Lab 3: Wi-Fi Security Assessment} \\
Focuses on Aircrack-ng suite, WPA2 cracking, evil twin setup.

\vspace{0.2cm}

\noindent\textbf{Lab 4: Bluetooth/BLE Exploitation} \\
Focuses on GATT attacks, Ubertooth, gattacker, BLE recon.

\vspace{0.2cm}

\noindent\textbf{Lab 5: Mobile Dynamic Analysis} \\
Focuses on MobSF automation, Objection scripting, SSL pinning bypass.

\vspace{0.2cm}

\noindent\textbf{Lab 6: Cellular Network Recon} \\
Focuses on OpenBTS/IMSI catcher simulation, SS7 basics.

\vspace{0.2cm}

\noindent\textbf{Lab 7: Root Detection Bypass} \\
Focuses on MagiskHide, Frida anti-root scripts.

\vspace{0.2cm}

\noindent\textbf{Lab 8: Mobile Forensics Acquisition} \\
Focuses on Cellebrite/Grayshift, ADB backup analysis.

\vspace{0.2cm}

\noindent\textbf{Lab 9: Wireless IDS Deployment} \\
Focuses on Kismet, Snort wireless rules.

\vspace{0.2cm}

\noindent\textbf{Lab 10: Mobile Pentest Report} \\
Focuses on complete app/network assessment with remediation.

\vspace{0.2cm}
